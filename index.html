  x=x+y
  y=x-y
  x=x-y
  return x,y

def max_and_min_value(array):
  smallest=array[0]
  largest=array[0]
  for num in array:
      if num<smallest:
          smallest=num
      if num>largest:
          largest=num
  return largest,smallest

array=[555,68,90,12,44,56,435]

1.WAP to sort an array


def sortarray(myarray):
  for i in range(len(myarray)):
      for j in range(i+1,len(myarray)):
          if myarray[i]>myarray[j]:
              temp=myarray[i]
              myarray[i]=myarray[j]
              myarray[j]=temp
  return myarray

myarray=[45,555,7,9,0,2,3,5,100]
print(sortarray(myarray))
'''
output:
[0, 2, 3, 5, 7, 9, 45, 100, 555]
